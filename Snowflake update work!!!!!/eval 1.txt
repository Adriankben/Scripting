# Check if Snowflake ODBC Driver is installed (both 64-bit and 32-bit)
$odbc64 = Get-ItemProperty "HKLM:\SOFTWARE\ODBC\ODBCINST.INI\SnowflakeDSIIDriver" -ErrorAction SilentlyContinue
$odbc32 = Get-ItemProperty "HKLM:\SOFTWARE\WOW6432Node\ODBC\ODBCINST.INI\SnowflakeDSIIDriver" -ErrorAction SilentlyContinue

if ($odbc64 -or $odbc32) {
    Write-Output "Snowflake ODBC Driver is installed."
    
    if ($odbc64) {
        Write-Output "64-bit version: $($odbc64.DriverODBCVer)"
        Exit 1
    }
    
    if ($odbc32) {
        Write-Output "32-bit version: $($odbc32.DriverODBCVer)"
        Exit 1
    }
    
} else {
    Write-Output "Snowflake ODBC Driver is NOT installed."
    Exit 0
}
