Sysmon64 - u force

$ErrorActionPreference = "Stop"

$zipUrl    = "https://download.sysinternals.com/files/Sysmon.zip"
$tempRoot  = Join-Path $env:TEMP "SysmonUpdate"
$zipPath   = Join-Path $tempRoot "Sysmon.zip"
$extractTo = Join-Path $tempRoot "Extracted"

New-Item -ItemType Directory -Path $tempRoot -Force | Out-Null
New-Item -ItemType Directory -Path $extractTo -Force | Out-Null

Invoke-WebRequest -Uri $zipUrl -OutFile $zipPath -UseBasicParsing
Expand-Archive -Path $zipPath -DestinationPath $extractTo -Force

$sysmonExe = Join-Path $extractTo "Sysmon64.exe"

if (-not (Test-Path $sysmonExe)) {
    throw "Sysmon64.exe not found: $sysmonExe"
}

& $sysmonExe -accepteula -i
