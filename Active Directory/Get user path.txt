# Import the Active Directory module
Import-Module ActiveDirectory

# Path to your CSV file
$csvPath = "C:\Path\To\initiallastnames.csv"

# Read each line as a username prefix (e.g., jsmith, ado)
$usernames = Get-Content $csvPath

# Prepare output
$results = @()

foreach ($name in $usernames) {
    $name = $name.Trim()

    # Search with wildcard match in SamAccountName
    $user = Get-ADUser -Filter "SamAccountName -like '$name*'" -Properties DisplayName, SamAccountName, DistinguishedName

    if ($user) {
        Write-Host "Found: $name"
        Write-Host "  DisplayName: $($user.DisplayName)"
        Write-Host "  SamAccountName: $($user.SamAccountName)"
        Write-Host "  DN: $($user.DistinguishedName)"
        Write-Host "---------------------------"

        $results += [PSCustomObject]@{
            InputName          = $name
            DisplayName        = $user.DisplayName
            SamAccountName     = $user.SamAccountName
            DistinguishedName  = $user.DistinguishedName
        }
    }
    else {
        Write-Host "Not found: $name"
        Write-Host "---------------------------"

        $results += [PSCustomObject]@{
            InputName          = $name
            DisplayName        = "Not Found"
            SamAccountName     = "Not Found"
            DistinguishedName  = "Not Found"
        }
    }
}

# Export to CSV
$results | Export-Csv -Path "C:\Path\To\AD_UserResults.csv" -NoTypeInformation
