# Define the application name (as it appears in Programs and Features)
$appName = "Dell Peripheral Manager"

# Registry paths where installed programs are listed
$registryPaths = @(
    "HKLM:\Software\Microsoft\Windows\CurrentVersion\Uninstall",
    "HKLM:\Software\Wow6432Node\Microsoft\Windows\CurrentVersion\Uninstall",
    "HKCU:\Software\Microsoft\Windows\CurrentVersion\Uninstall"
)

# Function to find the application uninstall string
function Get-UninstallString {
    param([string]$appName, [string[]]$paths)
    foreach ($path in $paths) {
        Get-ItemProperty -Path "$path\*" -ErrorAction SilentlyContinue |
        Where-Object { $_.DisplayName -eq $appName } |
        Select-Object -ExpandProperty UninstallString -ErrorAction SilentlyContinue
    }
}

# Find the uninstall string for the specified application
$uninstallString = Get-UninstallString -appName $appName -paths $registryPaths

if ($uninstallString) {
    Write-Host "Uninstalling $appName..."
    # Execute the uninstall command
    Start-Process -FilePath $uninstallString -ArgumentList "/quiet" -Wait
    Write-Host "$appName has been removed."
} else {
    Write-Host "Application '$appName' not found in Programs and Features."
}
