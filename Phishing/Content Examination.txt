Hash Commands – Verify file integrity and search threat intelligence using hashes (MD5, SHA1, SHA256). SHA256 is preferred for malware analysis due to lower collision risk.


Powershell
# MD5
Get-FileHash -Path "C:\path\to\file" -Algorithm MD5

# SHA1
Get-FileHash -Path "C:\path\to\file" -Algorithm SHA1

# SHA256
Get-FileHash -Path "C:\path\to\file" -Algorithm SHA256

CertUtil
# MD5
certutil -hashfile C:\path\to\file MD5

# SHA1
certutil -hashfile C:\path\to\file SHA1

# SHA256
certutil -hashfile C:\path\to\file SHA256


bash cli 
# MD5
md5sum /path/to/file

# SHA1
sha1sum /path/to/file

# SHA256
sha256sum /path/to/file

OpenSSL
# MD5
openssl dgst -md5 /path/to/file

# SHA1
openssl dgst -sha1 /path/to/file

# SHA256
openssl dgst -sha256 /path/to/file
